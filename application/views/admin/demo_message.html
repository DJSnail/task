#成功 alert-success 
                #失败 alert-danger
                #提示 alert-info
                $("#success").on('click',function() {
                    alert_message("操作成功","alert-success",true);
                });

                $("#fail").on('click',function() {
                     alert_message("操作失败",'',false);
                     return;
                });
                $("#warn").on('click',function() {
                     alert_message("警告：不能执行该操作","alert-info");
                     return;
                });
